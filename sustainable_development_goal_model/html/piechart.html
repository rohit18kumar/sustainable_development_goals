from matplotlib import pyplot as plt
_label=['<=5','5-10','10-15','15-20','20-25','25-30','30-35','>35']
_data=[2,9,8,4,3,1,5,4]
_color=['r','b','g','c','r','g','m','y']
plt.pie(_data,labels=_label,colors=_color,explode=[0.2,0,0.1,0,0.2,0,0.1,0],
        autopct=lambda p: '{:.0f}'.format(p * 36 / 100),startangle=45,shadow=True)
plt.title('Poverty SDG 1 Pie plot')
plt.show()
